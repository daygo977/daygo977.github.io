Example: FizzBuzz

FizzBuzz.pixkey
let i = 1
while i <= 100 do
    if i % 15 == 0 then
        print "FizzBuzz"
    else
        if i % 3  == 0 then
            print "Fizz"
        else
            if i % 5  == 0 then
                print "Buzz"
            else
                print i
            end
        end
    end
    i = i + 1
end
